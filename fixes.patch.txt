+def format_by_increment(value: Any, increment: Any) -> str:
+    """
+    Quantize `value` to the precision implied by `increment`.
+    Uses Decimal quantization built from the increment's exponent in a robust way.
+    """
+    try:
+        val = value if isinstance(value, Decimal) else Decimal(str(value))
+        inc = increment if isinstance(increment, Decimal) else Decimal(str(increment))
+        exponent = inc.normalize().as_tuple().exponent
+        # Build quantizer robustly (handles negative, zero, and typical exponents)
+        quant = Decimal(f"1e{exponent}")
+        quantized = val.quantize(quant, rounding=ROUND_DOWN)
+        decimals = abs(exponent)
+        return f"{quantized:.{decimals}f}"
+    except Exception as e:
+        logger.warning("format_by_increment failed for value=%s increment=%s: %s", value, increment, e)
+        return str(value)

+def _extract_path_segment(endpoint: str) -> str:
+    """
+    Safely extract the last path segment from an endpoint URL or path.
+    Handles query strings and trailing slashes.
+    """
+    try:
+        parsed = urlparse(endpoint)
+        path = parsed.path or endpoint
+        return path.rstrip("/").split("/")[-1]
+    except Exception:
+        return endpoint.split("/")[-1] if "/" in endpoint else endpoint

+class ConnectionManager:
+    """
+    Lightweight skeleton to manage primary/standby websocket clients and dedupe.
+    This is intentionally minimal — it provides a processed-event LRU and simple
+    helpers to mark events as seen. Use this as the basis for hot-standby promotion.
+    """
+    def __init__(self, max_cache: int = 5000):
+        self.processed = collections.OrderedDict()
+        self.max_cache = max_cache
+
+    def seen(self, key: str) -> bool:
+        return key in self.processed
+
+    def mark_seen(self, key: str):
+        try:
+            self.processed[key] = True
+            if len(self.processed) > self.max_cache:
+                self.processed.popitem(last=False)
+        except Exception:
+            # defensive: don't let dedupe failures crash the router
+            pass
+
